173 __STATIC_INLINE void nrf_lpcomp_configure(const nrf_lpcomp_config_t * p_config)
174 {
175  NRF_LPCOMP->TASKS_STOP = 1;
176  NRF_LPCOMP->ENABLE = LPCOMP_ENABLE_ENABLE_Disabled << LPCOMP_ENABLE_ENABLE_Pos;
177  NRF_LPCOMP->REFSEL =
178  (p_config->reference << LPCOMP_REFSEL_REFSEL_Pos) & LPCOMP_REFSEL_REFSEL_Msk;
179 
180  //If external source is choosen extract analog reference index.
181  if ((p_config->reference & LPCOMP_REFSEL_REFSEL_ARef)==LPCOMP_REFSEL_REFSEL_ARef)
182  {
183  uint32_t extref = p_config->reference >> 16;
184  NRF_LPCOMP->EXTREFSEL = (extref << LPCOMP_EXTREFSEL_EXTREFSEL_Pos) & LPCOMP_EXTREFSEL_EXTREFSEL_Msk;
185  }
186 
187  NRF_LPCOMP->ANADETECT =
188  (p_config->detection << LPCOMP_ANADETECT_ANADETECT_Pos) & LPCOMP_ANADETECT_ANADETECT_Msk;
189 #ifdef NRF52_SERIES
190  NRF_LPCOMP->HYST = ((p_config->hyst) << LPCOMP_HYST_HYST_Pos) & LPCOMP_HYST_HYST_Msk;
191 #endif
192  NRF_LPCOMP->SHORTS = 0;
193  NRF_LPCOMP->INTENCLR = LPCOMP_INTENCLR_CROSS_Msk | LPCOMP_INTENCLR_UP_Msk |
194  LPCOMP_INTENCLR_DOWN_Msk | LPCOMP_INTENCLR_READY_Msk;
195 }
