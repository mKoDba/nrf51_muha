626 __STATIC_INLINE void nrf_qspi_cinstrdata_set(NRF_QSPI_Type * p_reg,
627  nrf_qspi_cinstr_len_t length,
628  void const * p_tx_data)
629 {
630  uint32_t reg = 0;
631  uint8_t const *p_tx_data_8 = (uint8_t const *) p_tx_data;
632 
633  // Load custom instruction.
634  switch (length)
635  {
636  case NRF_QSPI_CINSTR_LEN_9B:
637  reg |= ((uint32_t)p_tx_data_8[7]) << QSPI_CINSTRDAT1_BYTE7_Pos;
638  /* fall-through */
639  case NRF_QSPI_CINSTR_LEN_8B:
640  reg |= ((uint32_t)p_tx_data_8[6]) << QSPI_CINSTRDAT1_BYTE6_Pos;
641  /* fall-through */
642  case NRF_QSPI_CINSTR_LEN_7B:
643  reg |= ((uint32_t)p_tx_data_8[5]) << QSPI_CINSTRDAT1_BYTE5_Pos;
644  /* fall-through */
645  case NRF_QSPI_CINSTR_LEN_6B:
646  reg |= ((uint32_t)p_tx_data_8[4]);
647  p_reg->CINSTRDAT1 = reg;
648  reg = 0;
649  /* fall-through */
650  case NRF_QSPI_CINSTR_LEN_5B:
651  reg |= ((uint32_t)p_tx_data_8[3]) << QSPI_CINSTRDAT0_BYTE3_Pos;
652  /* fall-through */
653  case NRF_QSPI_CINSTR_LEN_4B:
654  reg |= ((uint32_t)p_tx_data_8[2]) << QSPI_CINSTRDAT0_BYTE2_Pos;
655  /* fall-through */
656  case NRF_QSPI_CINSTR_LEN_3B:
657  reg |= ((uint32_t)p_tx_data_8[1]) << QSPI_CINSTRDAT0_BYTE1_Pos;
658  /* fall-through */
659  case NRF_QSPI_CINSTR_LEN_2B:
660  reg |= ((uint32_t)p_tx_data_8[0]);
661  p_reg->CINSTRDAT0 = reg;
662  /* fall-through */
663  case NRF_QSPI_CINSTR_LEN_1B:
664  /* Send only opcode. Case to avoid compiler warnings. */
665  break;
666  default:
667  break;
668  }
669 }

672  nrf_qspi_cinstr_len_t length,
673  void * p_rx_data)
674 {
675  uint8_t *p_rx_data_8 = (uint8_t *) p_rx_data;
676 
677  uint32_t reg = p_reg->CINSTRDAT1;
678  switch (length)
679  {
680  case NRF_QSPI_CINSTR_LEN_9B:
681  p_rx_data_8[7] = (uint8_t)(reg >> QSPI_CINSTRDAT1_BYTE7_Pos);
682  /* fall-through */
683  case NRF_QSPI_CINSTR_LEN_8B:
684  p_rx_data_8[6] = (uint8_t)(reg >> QSPI_CINSTRDAT1_BYTE6_Pos);
685  /* fall-through */
686  case NRF_QSPI_CINSTR_LEN_7B:
687  p_rx_data_8[5] = (uint8_t)(reg >> QSPI_CINSTRDAT1_BYTE5_Pos);
688  /* fall-through */
689  case NRF_QSPI_CINSTR_LEN_6B:
690  p_rx_data_8[4] = (uint8_t)(reg);
691  /* fall-through */
692  default:
693  break;
694  }
695 
696  reg = p_reg->CINSTRDAT0;
697  switch (length)
698  {
699  case NRF_QSPI_CINSTR_LEN_5B:
700  p_rx_data_8[3] = (uint8_t)(reg >> QSPI_CINSTRDAT0_BYTE3_Pos);
701  /* fall-through */
702  case NRF_QSPI_CINSTR_LEN_4B:
703  p_rx_data_8[2] = (uint8_t)(reg >> QSPI_CINSTRDAT0_BYTE2_Pos);
704  /* fall-through */
705  case NRF_QSPI_CINSTR_LEN_3B:
706  p_rx_data_8[1] = (uint8_t)(reg >> QSPI_CINSTRDAT0_BYTE1_Pos);
707  /* fall-through */
708  case NRF_QSPI_CINSTR_LEN_2B:
709  p_rx_data_8[0] = (uint8_t)(reg);
710  /* fall-through */
711  case NRF_QSPI_CINSTR_LEN_1B:
712  /* Send only opcode. Case to avoid compiler warnings. */
713  break;
714  default:
715  break;
