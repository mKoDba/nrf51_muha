306 __STATIC_INLINE void nrf_spi_configure(NRF_SPI_Type * p_reg,
307  nrf_spi_mode_t spi_mode,
308  nrf_spi_bit_order_t spi_bit_order)
309 {
310  uint32_t config = (spi_bit_order == NRF_SPI_BIT_ORDER_MSB_FIRST ?
311  SPI_CONFIG_ORDER_MsbFirst : SPI_CONFIG_ORDER_LsbFirst);
312  switch (spi_mode)
313  {
314  default:
315  case NRF_SPI_MODE_0:
316  config |= (SPI_CONFIG_CPOL_ActiveHigh << SPI_CONFIG_CPOL_Pos) |
317  (SPI_CONFIG_CPHA_Leading << SPI_CONFIG_CPHA_Pos);
318  break;
319 
320  case NRF_SPI_MODE_1:
321  config |= (SPI_CONFIG_CPOL_ActiveHigh << SPI_CONFIG_CPOL_Pos) |
322  (SPI_CONFIG_CPHA_Trailing << SPI_CONFIG_CPHA_Pos);
323  break;
324 
325  case NRF_SPI_MODE_2:
326  config |= (SPI_CONFIG_CPOL_ActiveLow << SPI_CONFIG_CPOL_Pos) |
327  (SPI_CONFIG_CPHA_Leading << SPI_CONFIG_CPHA_Pos);
328  break;
329 
330  case NRF_SPI_MODE_3:
331  config |= (SPI_CONFIG_CPOL_ActiveLow << SPI_CONFIG_CPOL_Pos) |
332  (SPI_CONFIG_CPHA_Trailing << SPI_CONFIG_CPHA_Pos);
333  break;
334  }
335  p_reg->CONFIG = config;
336 }
