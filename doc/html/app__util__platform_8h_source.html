207 static __INLINE uint8_t current_int_priority_get(void)
208 {
209  uint32_t isr_vector_num = __get_IPSR() & IPSR_ISR_Msk ;
210  if (isr_vector_num > 0)
211  {
212  int32_t irq_type = ((int32_t)isr_vector_num - EXTERNAL_INT_VECTOR_OFFSET);
213  return (NVIC_GetPriority((IRQn_Type)irq_type) & 0xFF);
214  }
215  else
216  {
217  return APP_IRQ_PRIORITY_THREAD;
218  }
219 }

227 static __INLINE uint8_t privilege_level_get(void)
228 {
229 #if __CORTEX_M == (0x00U) || defined(_WIN32) || defined(__unix) || defined(__APPLE__)
230  /* the Cortex-M0 has no concept of privilege */
231  return APP_LEVEL_PRIVILEGED;
232 #elif __CORTEX_M == (0x04U)
233  uint32_t isr_vector_num = __get_IPSR() & IPSR_ISR_Msk ;
234  if (0 == isr_vector_num)
235  {
236  /* Thread Mode, check nPRIV */
237  int32_t control = __get_CONTROL();
238  return control & CONTROL_nPRIV_Msk ? APP_LEVEL_UNPRIVILEGED : APP_LEVEL_PRIVILEGED;
239  }
240  else
241  {
242  /* Handler Mode, always privileged */
243  return APP_LEVEL_PRIVILEGED;
244  }
245 #endif
246 }
