436 __STATIC_INLINE NRF_GPIO_Type * nrf_gpio_pin_port_decode(uint32_t * p_pin)
437 {
438  ASSERT(*p_pin < NUMBER_OF_PINS);
439 #if (GPIO_COUNT == 1)
440  // The oldest definition case
441  return NRF_GPIO;
442 #else
443  if (*p_pin < P0_PIN_NUM)
444  {
445  return NRF_P0;
446  }
447  else
448  {
449  *p_pin = *p_pin & (P0_PIN_NUM - 1);
450  return NRF_P1;
451  }
452 #endif
453 }

478 __STATIC_INLINE void nrf_gpio_cfg(
479  uint32_t pin_number,
480  nrf_gpio_pin_dir_t dir,
481  nrf_gpio_pin_input_t input,
482  nrf_gpio_pin_pull_t pull,
483  nrf_gpio_pin_drive_t drive,

490  | ((uint32_t)pull << GPIO_PIN_CNF_PULL_Pos)
491  | ((uint32_t)drive << GPIO_PIN_CNF_DRIVE_Pos)
492  | ((uint32_t)sense << GPIO_PIN_CNF_SENSE_Pos);
