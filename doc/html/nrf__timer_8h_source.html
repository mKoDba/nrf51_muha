574 __STATIC_INLINE uint32_t nrf_timer_us_to_ticks(uint32_t time_us,
575  nrf_timer_frequency_t frequency)
576 {
577  // The "frequency" parameter here is actually the prescaler value, and the
578  // timer runs at the following frequency: f = 16 MHz / 2^prescaler.
579  uint32_t prescaler = (uint32_t)frequency;
580  ASSERT(time_us <= (UINT32_MAX / 16UL));
581  return ((time_us * 16UL) >> prescaler);
582 }
583 
584 __STATIC_INLINE uint32_t nrf_timer_ms_to_ticks(uint32_t time_ms,
585  nrf_timer_frequency_t frequency)
586 {
587  // The "frequency" parameter here is actually the prescaler value, and the
588  // timer runs at the following frequency: f = 16000 kHz / 2^prescaler.
589  uint32_t prescaler = (uint32_t)frequency;
590  ASSERT(time_ms <= (UINT32_MAX / 16000UL));
591  return ((time_ms * 16000UL) >> prescaler);
592 }
