472 __STATIC_INLINE void nrf_spis_configure(NRF_SPIS_Type * p_reg,
473  nrf_spis_mode_t spi_mode,
474  nrf_spis_bit_order_t spi_bit_order)
475 {
476  uint32_t config = (spi_bit_order == NRF_SPIS_BIT_ORDER_MSB_FIRST ?
477  SPIS_CONFIG_ORDER_MsbFirst : SPIS_CONFIG_ORDER_LsbFirst);
478 
479  switch (spi_mode)
480  {
481  default:
482  case NRF_SPIS_MODE_0:
483  config |= (SPIS_CONFIG_CPOL_ActiveHigh << SPIS_CONFIG_CPOL_Pos) |
484  (SPIS_CONFIG_CPHA_Leading << SPIS_CONFIG_CPHA_Pos);
485  break;
486 
487  case NRF_SPIS_MODE_1:
488  config |= (SPIS_CONFIG_CPOL_ActiveHigh << SPIS_CONFIG_CPOL_Pos) |
489  (SPIS_CONFIG_CPHA_Trailing << SPIS_CONFIG_CPHA_Pos);
490  break;
491 
492  case NRF_SPIS_MODE_2:
493  config |= (SPIS_CONFIG_CPOL_ActiveLow << SPIS_CONFIG_CPOL_Pos) |
494  (SPIS_CONFIG_CPHA_Leading << SPIS_CONFIG_CPHA_Pos);
495  break;
496 
497  case NRF_SPIS_MODE_3:
498  config |= (SPIS_CONFIG_CPOL_ActiveLow << SPIS_CONFIG_CPOL_Pos) |
499  (SPIS_CONFIG_CPHA_Trailing << SPIS_CONFIG_CPHA_Pos);
500  break;
501  }
502  p_reg->CONFIG = config;
503 }
