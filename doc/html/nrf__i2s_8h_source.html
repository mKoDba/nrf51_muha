443 __STATIC_INLINE bool nrf_i2s_configure(NRF_I2S_Type * p_i2s,
444  nrf_i2s_mode_t mode,
445  nrf_i2s_format_t format,
446  nrf_i2s_align_t alignment,
447  nrf_i2s_swidth_t sample_width,
448  nrf_i2s_channels_t channels,
449  nrf_i2s_mck_t mck_setup,
450  nrf_i2s_ratio_t ratio)
451 {
452  if (mode == NRF_I2S_MODE_MASTER)
453  {
454  // The MCK/LRCK ratio shall be a multiple of 2 * sample width.
455  if (((sample_width == NRF_I2S_SWIDTH_16BIT) &&
456  (ratio == NRF_I2S_RATIO_48X))
457  ||
458  ((sample_width == NRF_I2S_SWIDTH_24BIT) &&
459  ((ratio == NRF_I2S_RATIO_32X) ||
460  (ratio == NRF_I2S_RATIO_64X) ||
461  (ratio == NRF_I2S_RATIO_128X) ||
462  (ratio == NRF_I2S_RATIO_256X) ||
463  (ratio == NRF_I2S_RATIO_512X))))
464  {
465  return false;
466  }
467  }
468 
469  p_i2s->CONFIG.MODE = mode;
470  p_i2s->CONFIG.FORMAT = format;
471  p_i2s->CONFIG.ALIGN = alignment;
472  p_i2s->CONFIG.SWIDTH = sample_width;
473  p_i2s->CONFIG.CHANNELS = channels;
474  p_i2s->CONFIG.RATIO = ratio;
475 
476  if (mck_setup == NRF_I2S_MCK_DISABLED)
477  {
478  p_i2s->CONFIG.MCKEN =
479  (I2S_CONFIG_MCKEN_MCKEN_Disabled << I2S_CONFIG_MCKEN_MCKEN_Pos);
480  }
481  else
482  {
483  p_i2s->CONFIG.MCKFREQ = mck_setup;
484  p_i2s->CONFIG.MCKEN =
485  (I2S_CONFIG_MCKEN_MCKEN_Enabled << I2S_CONFIG_MCKEN_MCKEN_Pos);
486  }
487 
488  return true;
489 }
